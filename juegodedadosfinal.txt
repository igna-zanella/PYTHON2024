import random
cant=int(input("indique cantidad de jugadores"))
puntajes=[]
nombres=[]
seguir=[]
for i in range(cant):
    nombre=input("ingrese nombre")
    nombres.append(nombre)
    print("Juega el jugador :",nombre)
    puntaje=0
    dado=random.randint(1,6)
    print("Dado 1:",dado)
    puntaje+=dado
    dado=random.randint(1,6)
    print("Dado 2:",dado)
    puntaje+=dado
    dado=random.randint(1,6)
    print("Dado 3:",dado)
    puntaje+=dado
    print("Puntaje",puntaje)
    puntajes.append(puntaje)
    seguir.append(True)

flag=0
while flag<cant:
    for i in range(cant):
        if seguir[i]==True:
            if puntajes[i]>18:
                print("te pasaste")
                flag+=1
                seguir[i]=False
            else:
                print("Puntaje de",nombres[i],puntajes[i])
                print("Quiere otro dado?",nombres[i])
                opc=input()
                if opc=="n":
                    flag+=1
                    seguir[i]=False
                if seguir[i]==True:
                    dado=random.randint(1,6)
                    print(dado)
                    puntajes[i]+=dado
                    print("Puntaje de",nombres[i],puntajes[i])
puntmax=0
for i in range(cant):
    if puntajes[i]>puntmax and puntajes[i]<=18:
        ganadores=[]
        puntmax=puntajes[i]
        ganadores.append(i)
    elif puntajes[i]==puntmax:
        ganadores.append(i)
    print(ganadores)
        
if puntmax==0:
    print("Se pasaron todos")
else:
    if len(ganadores)>1:
        print("Hay un empate")
        cad="Los ganadores son:"
        for indice in ganadores:
            cad+=nombres[indice]
            cad+=","
        print(cad[:-1],"con un puntaje de",puntmax)
    else:
        print("Ganador:",nombres[ganadores[0]],puntajes[ganadores[0]])
for i in range(cant):
    print(nombres[i],puntajes[i])
