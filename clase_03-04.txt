else:
    print("La división no es exacta. Cociente:", '\033[33m'+str(cociente)+'\033[0m' ,"Resto:" , '\033[33m'+str(resto)+'\033[0m' ,".")

#  Ejercicio 8

# Escriba un programa que pida el año actual y un año cualquiera y que escriba cuántos años han pasado desde ese año o cuántos años faltan para llegar a ese año.

anioActual = int(input("¿En qué año estamos?: "))
anioVar = int(input("Escriba un año cualquiera: "))

if anioActual > anioVar:
    print("Desde el año", anioVar, "han pasado", anioActual - anioVar, "años.")
elif anioActual < anioVar:
    print("Para llegar al año", anioVar, "faltan", anioVar - anioActual, "años.")
else:
    print("¡Son el mismo año!")
